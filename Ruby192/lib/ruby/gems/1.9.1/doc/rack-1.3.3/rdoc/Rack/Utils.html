Rack::Utils contains a grab-bag of useful methods for writing web applications adopted from all kinds of Ruby libraries.

Parses the “Range:” header, if present, into an array of Range objects. Returns nil if the header is missing or syntactically invalid. Returns an empty array if none of the ranges are satisfiable. 

     # File lib/rack/utils.rb, line 284
284:     def byte_ranges(env, size)
285:       # See <http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.35>
286:       http_range = env['HTTP_RANGE']
287:       return nil unless http_range
288:       ranges = []
289:       http_range.split(/,\s*/).each do |range_spec|
290:         matches = range_spec.match(/bytes=(\d*)-(\d*)/)
291:         return nil  unless matches
292:         r0,r1 = matches[1], matches[2]
293:         if r0.empty?
294:           return nil  if r1.empty?
295:           # suffix-byte-range-spec, represents trailing suffix of file
296:           r0 = [size - r1.to_i, 0].max
297:           r1 = size - 1
298:         else
299:           r0 = r0.to_i
300:           if r1.empty?
301:             r1 = size - 1
302:           else
303:             r1 = r1.to_i
304:             return nil  if r1 < r0  # backwards range is syntactically invalid
305:             r1 = size-1  if r1 >= size
306:           end
307:         end
308:         ranges << (r0..r1)  if r0 <= r1
309:       end
310:       ranges
311:     end

rfc2822(time) click to toggle source

Modified version of stdlib time.rb Time#rfc2822 to use ’%d-%b-%Y’ instead of ’% %b %Y’. It assumes that the time is in GMT to comply to the RFC 2109.

NOTE: I’m not sure the RFC says it requires GMT, but is ambigous enough that I’m certain someone implemented only that option. Do not use %a and %b from Time.strptime, it would use localized names for weekday and month.

276:       mon = Time::RFC2822_MONTH_NAME[time.mon - 1]
277:       time.strftime("#{wday}, %d-#{mon}-%Y %H:%M:%S GMT")
278:     end

     # File lib/rack/utils.rb, line 163
163:     def select_best_encoding(available_encodings, accept_encoding)
164:       # http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html
165: 
166:       expanded_accept_encoding =
167:         accept_encoding.map { |m, q|
168:           if m == "*"
169:             (available_encodings - accept_encoding.map { |m2, _| m2 }).map { |m2| [m2, q] }
170:           else
171:             [[m, q]]
172:           end
173:         }.inject([]) { |mem, list|
174:           mem + list
175:         }
176: 
177:       encoding_candidates = expanded_accept_encoding.sort_by { |_, q| -q }.map { |m, _| m }
178: 
179:       unless encoding_candidates.include?("identity")
180:         encoding_candidates.push("identity")
181:       end
182: 
183:       expanded_accept_encoding.find_all { |m, q|
184:         q == 0.0
185:       }.each { |m, _|
186:         encoding_candidates.delete(m)
187:       }
188: 
189:       return (encoding_candidates & available_encodings)[0]
190:     end
set_cookie_header!(header, key, value) click to toggle source 
     # File lib/rack/utils.rb, line 193
193:     def set_cookie_header!(header, key, value)
194:       case value
195:       when Hash
196:         domain  = "; domain="  + value[:domain] if value[:domain]
197:         path    = "; path="    + value[:path]   if value[:path]
198:         # According to RFC 2109, we need dashes here.
199:         # N.B.: cgi.rb uses spaces...
200:         expires = "; expires=" +
201:           rfc2822(value[:expires].clone.gmtime) if value[:expires]
202:         secure = "; secure"  if value[:secure]
203:         httponly = "; HttpOnly" if value[:httponly]
204:         value = value[:value]
205:       end
206:       value = [value] unless Array === value
207:       cookie = escape(key) + "=" +
208:         value.map { |v| escape v }.join("&") +
209:         "#{domain}#{path}#{expires}#{secure}#{httponly}"
210: 
211:       case header["Set-Cookie"]
212:       when nil, ''
213:         header["Set-Cookie"] = cookie
214:       when String
215:         header["Set-Cookie"] = [header["Set-Cookie"], cookie].join("\n")
216:       when Array
217:         header["Set-Cookie"] = (header["Set-Cookie"] + [cookie]).join("\n")
218:       end
219: 
220:       nil
221:     end
