137: 
138:             lm = Win32::Registry::HKEY_LOCAL_MACHINE
139:             lm.open("SOFTWARE\\Mozilla\\Mozilla Firefox") do |reg|
140:               main = lm.open("SOFTWARE\\Mozilla\\Mozilla Firefox\\#{reg.keys[0]}\\Main")
